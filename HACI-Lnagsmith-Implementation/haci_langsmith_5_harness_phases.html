from langsmith import traceable
from typing import List, Optional
from pydantic import BaseModel, Field

class Hypothesis(BaseModel):
    """A hypothesis about the potential root cause."""
    description: str = Field(..., description="What we think might be causing the issue")
    confidence: float = Field(..., ge=0.0, le=1.0, description="Initial confidence 0-1")
    evidence_needed: List[str] = Field(default_factory=list, description="What we need to validate")
    validation_tools: List[str] = Field(default_factory=list, description="Tools to gather evidence")

class ThinkOutput(BaseModel):
    """Structured output from THINK phase."""
    hypotheses: List[Hypothesis]
    investigation_plan: List[str]
    reasoning: str
    next_tools: List[str]

@traceable(name="think_phase", metadata={"phase": "THINK"})
async def think_node(state: HarnessState) -> dict:
    """
    THINK phase: Analyze context and form hypotheses.
    
    This node receives the current state and produces:
    - Hypotheses ranked by likelihood
    - Investigation plan for validation
    - Tool calls to execute in ACT phase
    """
    
    # Build context from state
    context = build_think_context(state)
    
    # Previous observations inform new hypotheses
    prior_observations = state.get("observations", [])
    prior_hypotheses = state.get("hypotheses", [])
    
    # System prompt for THINK phase
    system_prompt = """You are an expert investigator in the THINK phase.

Your task is to analyze the available information and:
1. Form hypotheses about potential root causes
2. Rank hypotheses by likelihood based on evidence
3. Create a focused investigation plan
4. Identify specific tools needed to validate hypotheses

Consider:
- Previous observations that may support or refute hypotheses
- Correlations between symptoms and potential causes
- The severity and impact of the issue
- What evidence would definitively validate each hypothesis

Output your analysis in structured JSON format."""
    
    # Build user prompt with context
    user_prompt = format_think_prompt(
        ticket=state["ticket_description"],
        severity=state["ticket_severity"],
        prior_observations=prior_observations,
        prior_hypotheses=prior_hypotheses,
        iteration=state.get("iteration_count", 1)
    )
    
    # Call LLM with structured output
    llm = ChatAnthropic(model="claude-sonnet-4-20250514")
    structured_llm = llm.with_structured_output(ThinkOutput)
    
    result = await structured_llm.ainvoke([
        SystemMessage(content=system_prompt),
        HumanMessage(content=user_prompt)
    ])
    
    # Return state updates (uses Annotated reducers)
    return {
        "hypotheses": result.hypotheses,  # Appends to existing
        "investigation_plan": result.investigation_plan,
        "pending_tool_calls": result.next_tools,
        "current_phase": "ACT"
    }

from langsmith import traceable
from langchain_core.tools import BaseTool
import asyncio
import time

class ToolResult(BaseModel):
    """Result from a tool execution."""
    tool_name: str
    success: bool
    output: Any
    execution_time_ms: int
    metadata: dict = Field(default_factory=dict)

class ToolError(BaseModel):
    """Error from a tool execution."""
    tool_name: str
    error_type: str
    error_message: str
    recoverable: bool

@traceable(name="act_phase", metadata={"phase": "ACT"})
async def act_node(state: HarnessState) -> dict:
    """
    ACT phase: Execute tools to gather evidence.
    
    Key responsibilities:
    - Execute tools identified in THINK phase
    - Handle errors gracefully with fallbacks
    - Track execution metadata for observability
    - Rate limit and parallelize appropriately
    """
    
    pending_tools = state.get("pending_tool_calls", [])
    tool_registry = get_tool_registry()
    
    results: List[ToolResult] = []
    errors: List[ToolError] = []
    
    start_time = time.time()
    
    # Execute tools with controlled concurrency
    semaphore = asyncio.Semaphore(3)  # Max 3 concurrent tools
    
    async def execute_with_semaphore(tool_call: dict):
        async with semaphore:
            return await execute_single_tool(tool_call, tool_registry)
    
    # Run tools concurrently
    tasks = [execute_with_semaphore(tc) for tc in pending_tools]
    task_results = await asyncio.gather(*tasks, return_exceptions=True)
    
    for tool_call, result in zip(pending_tools, task_results):
        if isinstance(result, Exception):
            errors.append(ToolError(
                tool_name=tool_call["name"],
                error_type=type(result).__name__,
                error_message=str(result),
                recoverable=is_recoverable_error(result)
            ))
        else:
            results.append(result)
    
    total_time_ms = int((time.time() - start_time) * 1000)
    
    # Compile raw evidence for OBSERVE phase
    raw_evidence = {
        r.tool_name: r.output for r in results if r.success
    }
    
    return {
        "tool_results": results,
        "tool_errors": errors,
        "raw_evidence": raw_evidence,
        "execution_metadata": {
            "total_time_ms": total_time_ms,
            "tools_executed": len(results),
            "tools_failed": len(errors)
        },
        "current_phase": "OBSERVE"
    }

@traceable(name="execute_tool")
async def execute_single_tool(
    tool_call: dict,
    registry: dict[str, BaseTool]
) -> ToolResult:
    """Execute a single tool with timing and error handling."""
    
    tool_name = tool_call["name"]
    tool_args = tool_call.get("arguments", {})
    
    if tool_name not in registry:
        raise ValueError(f"Unknown tool: {tool_name}")
    
    tool = registry[tool_name]
    start = time.time()
    
    try:
        output = await tool.ainvoke(tool_args)
        return ToolResult(
            tool_name=tool_name,
            success=True,
            output=output,
            execution_time_ms=int((time.time() - start) * 1000),
            metadata={"args": tool_args}
        )
    except Exception as e:
        return ToolResult(
            tool_name=tool_name,
            success=False,
            output={"error": str(e)},
            execution_time_ms=int((time.time() - start) * 1000),
            metadata={"args": tool_args, "error_type": type(e).__name__}
        )

from langsmith import traceable

class Observation(BaseModel):
    """An insight extracted from evidence."""
    source_tool: str
    finding: str
    significance: str  # high, medium, low
    supports_hypotheses: List[str]
    refutes_hypotheses: List[str]

class Correlation(BaseModel):
    """A pattern observed across multiple data sources."""
    sources: List[str]
    pattern: str
    confidence: float
    implication: str

class ObserveOutput(BaseModel):
    """Structured output from OBSERVE phase."""
    observations: List[Observation]
    correlations: List[Correlation]
    hypothesis_confidence_updates: dict[str, float]
    gaps_identified: List[str]
    analysis_summary: str

@traceable(name="observe_phase", metadata={"phase": "OBSERVE"})
async def observe_node(state: HarnessState) -> dict:
    """
    OBSERVE phase: Analyze evidence and update hypotheses.
    
    Key responsibilities:
    - Extract insights from raw tool outputs
    - Identify correlations across data sources
    - Update hypothesis confidence based on evidence
    - Surface gaps requiring additional investigation
    """
    
    raw_evidence = state.get("raw_evidence", {})
    current_hypotheses = state.get("hypotheses", [])
    tool_errors = state.get("tool_errors", [])
    
    system_prompt = """You are an expert analyst in the OBSERVE phase.

Your task is to analyze the evidence gathered and:
1. Extract key observations from each tool's output
2. Identify correlations and patterns across sources
3. Update confidence scores for each hypothesis
4. Identify gaps - what information is still missing?

For each observation, note:
- Which hypothesis it supports or refutes
- The significance level (high/medium/low)
- Any temporal or causal relationships

Be precise and evidence-based. Avoid speculation beyond what the data shows.
Output your analysis in structured JSON format."""
    
    # Format evidence for analysis
    evidence_text = format_evidence_for_analysis(raw_evidence, tool_errors)
    hypotheses_text = format_hypotheses(current_hypotheses)
    
    user_prompt = f"""## Current Hypotheses
{hypotheses_text}

## Evidence Collected
{evidence_text}

## Tool Errors (if any)
{format_errors(tool_errors)}

Analyze this evidence and provide structured observations."""
    
    llm = ChatAnthropic(model="claude-sonnet-4-20250514")
    structured_llm = llm.with_structured_output(ObserveOutput)
    
    result = await structured_llm.ainvoke([
        SystemMessage(content=system_prompt),
        HumanMessage(content=user_prompt)
    ])
    
    # Apply confidence updates to hypotheses
    updated_hypotheses = apply_confidence_updates(
        current_hypotheses,
        result.hypothesis_confidence_updates
    )
    
    return {
        "observations": result.observations,  # Appends via reducer
        "correlations": result.correlations,
        "hypotheses": updated_hypotheses,
        "gaps": result.gaps_identified,
        "current_phase": "EVALUATE"
    }

def apply_confidence_updates(
    hypotheses: List[Hypothesis],
    updates: dict[str, float]
) -> List[Hypothesis]:
    """Apply confidence adjustments with Bayesian-style updates."""
    updated = []
    for h in hypotheses:
        if h.description in updates:
            # Blend prior and new evidence
            delta = updates[h.description]
            new_conf = min(1.0, max(0.0, h.confidence + delta))
            h.confidence = new_conf
        updated.append(h)
    return updated

from langsmith import traceable
from enum import Enum

class Decision(str, Enum):
    CONTINUE = "continue"
    COMPLETE = "complete"
    ESCALATE = "escalate"

class Finding(BaseModel):
    """Final investigation finding."""
    root_cause: str
    confidence: float
    supporting_evidence: List[str]
    recommended_actions: List[str]
    affected_systems: List[str]

class EvaluateOutput(BaseModel):
    """Structured output from EVALUATE phase."""
    decision: Decision
    overall_confidence: float
    confidence_breakdown: dict[str, float]
    finding: Optional[Finding] = None
    escalation_reason: Optional[str] = None
    next_investigation_focus: Optional[str] = None

@traceable(name="evaluate_phase", metadata={"phase": "EVALUATE"})
async def evaluate_node(state: HarnessState) -> dict:
    """
    EVALUATE phase: Make decision about investigation.
    
    Decision logic:
    1. Calculate overall confidence from hypotheses
    2. Check for conflicts or high-risk conditions
    3. Determine if completion criteria are met
    4. Generate finding if confident, or plan next iteration
    """
    
    hypotheses = state.get("hypotheses", [])
    observations = state.get("observations", [])
    gaps = state.get("gaps", [])
    iteration = state.get("iteration_count", 1)
    max_iterations = state.get("max_iterations", 5)
    confidence_threshold = state.get("confidence_threshold", 0.85)
    
    # Calculate confidence metrics
    top_hypothesis = get_highest_confidence_hypothesis(hypotheses)
    overall_confidence = calculate_weighted_confidence(hypotheses, observations)
    has_conflicts = detect_evidence_conflicts(observations)
    high_risk = is_high_risk_scenario(top_hypothesis, state["ticket_severity"])
    
    # Decision logic
    if high_risk:
        decision = Decision.ESCALATE
        reason = "High-risk scenario requires human approval"
    elif has_conflicts:
        decision = Decision.ESCALATE
        reason = "Conflicting evidence requires human judgment"
    elif overall_confidence >= confidence_threshold:
        decision = Decision.COMPLETE
        reason = None
    elif iteration >= max_iterations:
        decision = Decision.ESCALATE
        reason = f"Max iterations ({max_iterations}) reached without confidence"
    else:
        decision = Decision.CONTINUE
        reason = None
    
    # Generate finding if completing
    finding = None
    if decision == Decision.COMPLETE:
        finding = await generate_finding(top_hypothesis, observations)
    
    return {
        "decision": decision.value,
        "current_confidence": overall_confidence,
        "finding": finding,
        "escalation_reason": reason,
        "requires_human_approval": decision == Decision.ESCALATE,
        "iteration_count": iteration + 1,
        "current_phase": "ROUTER"
    }

def route_after_evaluate(state: HarnessState) -> str:
    """Conditional routing based on EVALUATE decision."""
    decision = state.get("decision")
    
    if decision == "complete":
        return "end"
    elif decision == "escalate":
        return "human_gate"
    else:  # continue
        return "think"  # Loop back to THINK

Loop Control

The EVALUATE phase controls the investigation loop. It increments the iteration counter and uses conditional edges to route back to THINK or terminate the graph.

State Flow Through Phases

1

Initial State Created

State initialized with ticket context, execution mode, empty hypotheses/observations lists, iteration_count=1, current_phase="THINK"

2

THINK Updates State

Adds hypotheses[] (via reducer), sets investigation_plan, pending_tool_calls, updates current_phase to "ACT"

3

ACT Updates State

Adds tool_results[], raw_evidence{}, execution_metadata, clears pending_tool_calls, updates current_phase to "OBSERVE"

4

OBSERVE Updates State

Adds observations[] (via reducer), correlations[], updates hypotheses with new confidence, identifies gaps, updates current_phase to "EVALUATE"

5

EVALUATE Makes Decision

Sets decision (continue/complete/escalate), current_confidence, finding (if complete), requires_human_approval, increments iteration_count

6

Router Directs Flow

Routes to: THINK (loop continues with accumulated state), END (graph completes with finding), or HUMAN_GATE (interrupts for approval)
